je_farms_invention_progress = {
    is_shown_when_inactive = {
        any_scope_state = {
            any_scope_building = {
                is_building_type = building_research_center
                has_active_production_method = pm_farms_research_funding
            }
        }
    }

    possible = {
        any_scope_state = {
            any_scope_building = {
                is_building_type = building_research_center
                has_active_production_method = pm_farms_research_funding
            }
        }
    }

    icon = "gfx/interface/icons/event_icons/event_industry.dds"

    group = je_group_funding

    scripted_progress_bar = invention_farms_progress_bar

    transferable = no

    should_be_pinned_by_default = no

    weight = 10000

    on_monthly_pulse = {
        effect = {
            if = {
                limit = {
                    scope:journal_entry = {
                        "scripted_bar_progress(invention_farms_progress_bar)" >= 100
                    }
                }
                if = {
                    limit = {
                        AND = {
                            NOT = {
                                has_technology_researched = scientific_crop_rotation
                            }
                            has_technology_researched = enclosure
                        }
                    }
                    add_technology_researched = scientific_crop_rotation
                    scope:journal_entry = {
                        add_progress = {
                            name = invention_farms_progress_bar
                            value = -100
                        }
                    }
                }
                else_if = {
                    limit = {
                        AND = {
                            NOT = {
                                has_technology_researched = mechanical_harvesting_systems
                            }
                            has_technology_researched = scientific_crop_rotation
                        }
                    }
                    add_technology_researched = mechanical_harvesting_systems
                    scope:journal_entry = {
                        add_progress = {
                            name = invention_farms_progress_bar
                            value = -100
                        }
                    }
                }
                else_if = {
                    limit = {
                        AND = {
                            NOT = {
                                has_technology_researched = hybrid_grain_cultivation
                            }
                            has_technology_researched = mechanical_harvesting_systems
                        }
                    }
                    add_technology_researched = hybrid_grain_cultivation
                    scope:journal_entry = {
                        add_progress = {
                            name = invention_farms_progress_bar
                            value = -100
                        }
                    }
                }
                else_if = {
                    limit = {
                        AND = {
                            NOT = {
                                has_technology_researched = motorized_irrigation_systems
                            }
                            has_technology_researched = hybrid_grain_cultivation
                        }
                    }
                    add_technology_researched = motorized_irrigation_systems
                    scope:journal_entry = {
                        add_progress = {
                            name = invention_farms_progress_bar
                            value = -100
                        }
                    }
                }

            }
