je_aga_main = {
	icon = "gfx/interface/icons/state_status_icons/split_state_icon.dds"
	group = je_group_historical_content

	scripted_progress_bar = aga_progress_bar
	complete = {
		custom_tooltip = {
			text = textaga
			has_variable = ref_compl
		}
	}
	fail = {
		custom_tooltip = {
			text = aga_tool1
			has_variable = var_agam
		}
	}
	on_fail = {
		trigger_event = {
			id = fm_aga.2
		}
		if = {
			limit = {
				has_modifier = diplomatic_plays_blocked
			}
			remove_modifier = diplomatic_plays_blocked
		}
	}
	should_be_pinned_by_default = yes

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					scope:journal_entry = { 
						"scripted_bar_progress(aga_progress_bar)" <= 1
					}
				 }
				 set_variable = {
					name = var_agam
					days = 30
				}
			}
		}
	}
}

je_aga_reforms = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_historical_content
	timeout = 1500

	complete = {
		has_law = law_type:law_slavery_banned
		OR = {
			NOT = {
				has_law = law_type:law_oligarchy
			}
			NOT = {
				has_law = law_type:law_hereditary_bureaucrats
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = fm_aga.3
		}
		remove_modifier = diplomatic_plays_blocked
		set_variable = {
			name = ref_compl
			days = 30
		}
	}

	on_timeout = {
		trigger_event = {
			id = fm_aga.4
		}
		if = {
			limit = {
				has_modifier = diplomatic_plays_blocked
			}
			remove_modifier = diplomatic_plays_blocked
		}
	}
	transferable = yes

	weight = 10000
	should_be_pinned_by_default = yes
}
