je_dakota_main = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	group = je_group_historical_content
	immediate = {
		set_variable = {
			name = dakota_progress
			value = 0
		}
	}
	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}
	on_complete = {
		ig:ig_armed_forces = {
			remove_ideology = ideology_indian
			add_ideology = ideology_jingoist
		}
		ig:ig_landowners = {
			remove_ideology = ideology_indian
			add_ideology = ideology_stratocratic
		}
		ig:ig_rural_folk = {
			if = {
				limit = {
					has_ideology = ideology:ideology_indian
				}
				remove_ideology = ideology_indian
				add_ideology = ideology_isolationist
				add_ideology = ideology_agrarian
			}
		}
		add_modifier = {
			name = modifier_dak_two
			months = 190
			is_decaying = yes
		}
		if = {
			limit = {
				has_modifier = modifier_dak_one
			}
			remove_modifier = modifier_dak_one
		}
		set_variable = {
			name = je_dak_compl
		}
	}
	on_monthly_pulse = {
		effect = {
			# Wargoal enforced scores are handled in the on_wargoal_enforced on_action
			# Advance Conditions
			if = {
				limit = {
					NOT = {
						has_law = law_type:law_isolationism
					}
				}
				change_variable = {
					name = dakota_progress
					add = 1
				}
			}
			if = {
				limit = {
					NOT = {
						has_law = law_type:law_industry_banned
					}
				}
				change_variable = {
					name = dakota_progress
					add = 1
				}
			}
			if = {
				limit = {
					has_technology_researched = line_infantry
				}
				change_variable = {
					name = dakota_progress
					add = 1
				}
			}
			if = {
				limit = {
					# Literacy rating is high-isch
					literacy_rate >= 0.3
				}
				change_variable = {
					name = dakota_progress
					add = 1
				}
			}
			if = {
				limit = {
					any_scope_building = {
						is_building_type = building_university
						occupancy > 0.1
						level >= 1
					}
				}
				change_variable = {
					name = dakota_progress
					add = 1
				}
			}
			if = {
				limit = {
					NOT = {
						has_law = law_type:law_closed_borders
					}
				}
				change_variable = {
					name = dakota_progress
					add = 1
				}
			}
			clamp_variable = {
				name = dakota_progress
				max = 200
				min = 0
			}
		}
	}
	current_value = {
		value = root.var:dakota_progress
	}
	goal_add_value = {
		value = 200
	}
	progressbar = yes
	weight = 100
	should_be_pinned_by_default = yes
}
