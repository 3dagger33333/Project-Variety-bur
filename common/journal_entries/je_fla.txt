			ruler ?= { #TODO: Ruler/Heir missing

			}
			heir ?= {
				set_character_immortal = yes
			}
		}
		c:FLA = {
			ruler ?= {
				set_character_immortal = yes
			}
		}
	}

	complete = {
		OR = {
			AND = {
				c:BUD = {

				}

					is_power_bloc_leader = yes
				}
			}
			AND = {
				c:BUD = {
					is_subject = no
					relations:ROOT >= relations_threshold:amicable
					is_ai = yes
				}
				cultural_acceptance_base = { target = cu:burgundian value >= 50 }
			}
		}
	}
	fail = {
		OR = {
			c:BUD = {
                is_subject_of = ROOT
            }
			custom_tooltip = {
				text = nobud

					exists = c:BUD
				}
			}
			custom_tooltip = {
				text = flamarr

					has_variable = is_married
				}
			}
		}
	}

		create_diplomatic_pact = {
			country = c:BUD
			type = personal_union
		}
		trigger_event = {
			id = fm_fla.1
		}
		hidden_effect = {
			c:BUD = {
				ruler = {
					set_character_immortal = no
				}
				heir = {
					set_character_immortal = no
				}

					name = var_matr
				}
			}

				ruler = {
					set_character_immortal = no
				}
			}
		}
	}
	on_fail = {

			c:BUD = {
				ruler = {
					set_character_immortal = no
				}
				heir = {
					set_character_immortal = no
				}
			}
			c:FLA = {
				ruler = {
					set_character_immortal = no
				}
			}
		}
	}
	on_timeout = {

			c:BUD = {
				ruler = {
					set_character_immortal = no
				}
				heir = {
					set_character_immortal = no
				}
			}
			c:FLA = {
				ruler = {
					set_character_immortal = no
				}
			}
