je_fru_main = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	group = je_group_historical_content
	immediate = {
		set_variable = {
			name = progress_fra
			value = 0
		}
	}
	scripted_button = je_fru_orl
	scripted_button = je_fru_anj
	scripted_button = je_fru_poi
	complete = {
			scope:journal_entry = {
				is_goal_complete = yes
			}
	}
	fail = {
		NOT = {
			has_law = law_type:law_monarchy
		}
	}
	on_complete = {
		trigger_event = {
			id = fm_fru.2
		}
		remove_variable = progress_fra
		hidden_effect = {
		annex_with_incorporation = c:POI
		annex_with_incorporation = c:ORL
		annex_with_incorporation = c:ANJ		
		}

	}
	on_fail = {
		remove_variable = progress_fra
	}
	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					has_law = law_type:law_monarchy
				}
				change_variable = {
					name = progress_fra
					add = 1
				}
			}
			if = {
				limit = {
					has_law = law_type:law_autocracy
				}
				change_variable = {
					name = progress_fra
					add = 3
				}
			}
			if = {
				limit = {
					has_law = law_type:law_appointed_bureaucrats
				}
				change_variable = {
					name = progress_fra
					add = 1
				}
			}
			if = {
				limit = {
					has_law = law_type:law_professional_army
				}
				change_variable = {
					name = progress_fra
					add = 1
				}
			}
			if = {
				limit = {
					has_modifier = modifier_fru_four
				}
				change_variable = {
					name = progress_fra
					add = 1
				}
			}
			if = {
				limit = {
					has_modifier = modifier_fru_five
				}
				change_variable = {
					name = progress_fra
					add = 1
				}
			}
			if = {
				limit = {
					has_modifier = modifier_fru_six
				}
				change_variable = {
					name = progress_fra
					add = 1
				}
			}
			clamp_variable = {
				name = progress_fra
				max = 100
				min = 0
			}
		}
	}

	current_value = {
		value = root.var:progress_fra
	}
	goal_add_value = {
		value = 100
	}
	progressbar = yes
	weight = 100
	should_be_pinned_by_default = yes
}
