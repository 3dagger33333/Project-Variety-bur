	icon = "gfx/interface/icons/law_icons/hereditary_bureaucrats.dds"
	
	group = je_group_historical_content

	complete = {
		ig:ig_landowners = {
			is_powerful = no
		}
		NOT = { has_law = law_type:law_serfdom }
		NOT = { has_law = law_type:law_hereditary_bureaucrats }
		NOT = { has_law = law_type:law_local_police }
		NOT = { has_law = law_type:law_peasant_levies }
	}

	on_complete = {
		custom_tooltip = {
			text = progressjapmod

				name = jaep_var

		}

				set_interest_group_name = ig_landowners
			}	
		}
	}


	weight = 10000
	should_be_pinned_by_default = yes
}

je_jaep_economy = {
	icon = "gfx/interface/icons/event_icons/event_industry.dds"
	
	group = je_group_historical_content

	complete = {
		in_default = no
		any_scope_state = {
			filter = {
				is_incorporated = yes
			}

				is_building_type = building_urban_center
				level >= 5

		custom_tooltip = {
			text = progressjapmod

	should_be_pinned_by_default = yes

}

je_jaep_army = {
	icon = "gfx/interface/ve_national_ideas/japan_idea2.dds"
	
	group = je_group_historical_content
	scripted_button = je_jae_button

	complete = {
		OR = {
			custom_tooltip = {
				text = yessamu
				has_variable = var_samuyes
			}
			AND = {
				ig:ig_armed_forces = {
					is_in_government = no
				}
				has_technology_researched = napoleonic_warfare
				NOT = {
					any_military_formation = {
						filter = {
							is_army = yes
						}
						any_combat_unit = {
							has_unit_type = unit_type:combat_unit_type_irregular_infantry
						}
						percent >= 0.25
					}
				}
			}
		}
	}

	on_complete = {
		custom_tooltip = {
			text = progressjapmod
			change_variable = {
				name = jaep_var
				add = 1
			}
		}
		if = {
			limit = { 
				NOT = {
					has_variable = var_samuyes
				}
			}
			trigger_event = {
				id = jaep.3
			}
		}
	}

	on_monthly_pulse = {
		random_events = {
			100 = 0
			10 = jaep.9
			10 = jaep.10
		}
	}

	transferable = no

	weight = 5000
	should_be_pinned_by_default = yes
}
