yag_progress_bar = {
	name = "yag_progress_bar"
	desc = "yag_progress_bar_desc"

	default = yes

	monthly_progress = {
		if = {
			limit = {
				has_technology_researched = standing_army 
			}
			add = {
				value = -5
			}
		}
		if = {
			limit = {
				OR = {
					is_enacting_law = law_type:law_autocracy
					is_enacting_law = law_type:law_single_party_state
					is_enacting_law = law_type:law_landed_voting
					is_enacting_law = law_type:law_wealth_voting
					is_enacting_law = law_type:law_anarchy
					is_enacting_law = law_type:law_census_voting
					is_enacting_law = law_type:law_universal_suffrage
					is_enacting_law = law_type:law_technocracy
				}
			}
			add = {
				value = -40
			}
		}
	}

	start_value = 210
	min_value = 0
	max_value = 500

 }

 aga_progress_bar = {
	name = "aga_progress_bar"
	desc = "aga_progress_bar_desc"

	default = yes
	is_inverted = yes

	monthly_progress  = {
    if = {
			limit = {
				any_scope_state = {
					owner = root
					is_incorporated = yes
					turmoil >= 0.3
					count = 1
				}
			}
			add = {
				value = -1
			}
		}
		if = {
			limit = {
				any_scope_state = {
					owner = root
					is_incorporated = yes
					turmoil >= 0.3
					count = 2
				}
			}
			add = {
				value = -2
			}
		}
		if = {
			limit = {
				any_scope_state = {
					owner = root
					is_incorporated = yes
					turmoil >= 0.3
					count = 3
				}
			}
			add = {
				value = -3
			}
		}
		if = {
			limit = {
				any_scope_state = {
					owner = root
					is_incorporated = yes
					turmoil >= 0.3
					count = 4
				}
			}
			add = {
				value = -4
			}
		}
		if = {
			limit = {
				any_scope_state = {
					owner = root
					is_incorporated = yes
					turmoil >= 0.3
					count > 5
				}
			}
			add = {
				value = -5
			}
		}
	}

	start_value = 199
	min_value = 0
	max_value = 200

 }
