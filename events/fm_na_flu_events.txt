	}

	icon = "gfx/interface/icons/event_icons/event_default.dds"

	option = {
		name = fm_na_flu.3.a
		default_option = yes
		add_modifier = {
			name = modifier_flu_no_cuba

			is_decaying = yes

			limit = {

			}

					target = c:CAS
				}
			}
		}
	}
}
