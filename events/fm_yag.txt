	event_image = {
		video = "unspecific_politicians_arguing"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	trigger = {
		any_scope_state = {
			count >= 0
			AND = {
				region = sr:region_oceania
				owner = c:SAU
			}
		}
		NOT = {
			has_variable = yag_demands_event
		}
	}

	immediate = {
		set_variable = yag_demands_event
	}

	option = {
		name = fm_yag.7.a
		c:SAU = {
			create_diplomatic_catalyst = {  
				type = catalyst_event_negative
				target = c:YAG
			}
		}
		change_relations = {
			country = c:SAU
			value = -20
		}
		s:STATE_NORTHERN_TERRITORY = {
			add_claim = root
		}
		add_journal_entry = {
			type = je_yag_sec
		}
	}

	option = {
		name = fm_yag.7.b
		custom_tooltip = {
			text = var_small_app
			je:je_yag_main = {
				add_progress = { value = -15 name = yag_progress_bar }
			}
		}
		change_relations = {
			country = c:WTI
			value = -10
		}
		change_relations = {
			country = c:MRN
			value = -10
		}
		change_relations = {
			country = c:NNG
			value = -10
		}
		change_relations = {
			country = c:WAT

		}

			add_claim = root
		}
	}
